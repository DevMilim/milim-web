1use std::{collections::HashMap, io::Result};
2
3#[derive(Clone, Debug, PartialEq)]
4pub struct HttpResponse {
5    pub version: String,
6    pub status_code: String,
7    pub status_text: String,
8    pub headers: Option<HashMap<String, String>>,
9    pub body: Option<String>,
10}
11
12impl Default for HttpResponse {
13    fn default() -> Self {
14        Self {
15            version: "HTTP/1.1".to_string(),
16            status_code: "200".to_string(),
17            status_text: "OK".to_string(),
18            headers: None,
19            body: None,
20        }
21    }
22}
23
24impl HttpResponse {
25    pub fn new(
26        status_code: &str,
27        headers: Option<HashMap<String, String>>,
28        body: Option<String>,
29    ) -> Self {
30        let mut response: HttpResponse = HttpResponse::default();
31        if status_code != "200" {
32            response.status_code = status_code.into();
33        }
34        response.headers = match &headers {
35            Some(_) => headers,
36            None => {
37                let mut h = HashMap::new();
38                h.insert("Content-Type".to_string(), "text/html".to_string());
39                Some(h)
40            }
41        };
42        response.status_text = match response.status_code.as_str() {
43            "200" => "OK".to_string(),
44            "400" => "Bad Request".to_string(),
45            "404" => "Not Found".to_string(),
46            "405" => "Method not Allowed".to_string(),
47            "500" => "Internal Server Error".to_string(),
48            _ => "Not Found".to_string(),
49        };
50        response.body = body;
51        response
52    }
53
54    pub fn get_version(&self) -> &str {
55        &self.version
56    }
57    pub fn get_status_code(&self) -> &str {
58        &self.status_code
59    }
60    pub fn get_status_text(&self) -> &str {
61        &self.status_text
62    }
63    pub fn get_headers(&self) -> String {
64        let map: HashMap<String, String> = self.headers.clone().expect("Erro ao obter headers");
65        let mut header_string: String = "".into();
66        for (k, v) in map.iter() {
67            header_string = format!("{}{}:{}\r\n", header_string, k, v);
68        }
69        header_string
70    }
71    pub fn get_body(&self) -> &str {
72        match &self.body {
73            Some(b) => b.as_str(),
74            None => "",
75        }
76    }
77    pub fn body(&mut self, body: &str) -> Self {
78        self.body = Some(body.to_string());
79        self.clone()
80    }
81    pub fn add_header(&mut self, key: &str, value: &str) -> Result<()> {
82        let mut h = match self.headers.clone() {
83            Some(headers) => headers,
84            None => HashMap::new(),
85        };
86        h.insert(key.to_string(), value.to_string());
87        self.headers = Some(h);
88        Ok(())
89    }
90    pub fn remove_header(&mut self, key: &str) -> Result<()> {
91        let mut h = match self.headers.clone() {
92            Some(headers) => headers,
93            None => HashMap::new(),
94        };
95        h.remove(key);
96        self.headers = Some(h);
97        Ok(())
98    }
99}

101impl From<HttpResponse> for String {
102    fn from(value: HttpResponse) -> Self {
103        let res = value.clone();
104        format!(
105            "{} {} {}\r\n{}Content-Length: {}\r\n\r\n{}",
106            &res.get_version(),
107            &res.get_status_code(),
108            &res.get_status_text(),
109            &res.get_headers(),
110            &res.get_body().len(),
111            &res.get_body()
112        )
113    }

50                Ok(mut stream) => {

52                }

59fn handle_connection(
60    routes: &[(String, Handler)],
61    stream: &mut TcpStream,
62    ctx: &Context,
63) -> Result<()> {
64    let mut buf = [0u8; 8192];
65    let n = stream.read(&mut buf).expect("Erro ao ler stream");
66    if n == 0 {
67        return Ok(());
68    }
69
70    let raw = String::from_utf8_lossy(&buf[..n]).to_string();
71    println!("Raw request: \n{}", raw);
72
73    let mut req = HttpRequest::from(raw);
74    println!("{:#?}", req);
75    let path = match &req.resource {
76        Resource::Path(p) => p.clone(),
77    };
78
79    for (pattern, handler) in routes {
80        if let Some(params) = match_route(pattern, &path) {
81            req.params = Some(params.0);
82            req.queryes = Some(params.1);
83
84            let res_string: String = handler(&req, ctx).response().into();
85            stream
86                .write_all(res_string.as_bytes())
87                .expect("Erro ao escrever buffer de resposta");
88            return Ok(());
89        }
90    }
91    Ok(())
92}

94fn split_path_query(s: &str) -> (&str, Option<&str>) {
95    if let Some(pos) = s.find("?") {
96        (&s[..pos], Some(&s[pos + 1..]))
97    } else {
98        (&s, None)
99    }
100}
101fn parse_query(q: Option<&str>) -> HashMap<String, String> {
102    let mut map = HashMap::new();
103    if let Some(qs) = q {
104        for pair in qs.split("&") {
105            if pair.is_empty() {
106                continue;
107            }
108            let mut it = pair.splitn(2, "=");
109            let key = it.next().unwrap_or("");
110            let val = it.next().unwrap_or("");
111            map.insert(key.to_string(), val.to_string());
112        }
113    }
114    map
115}

121fn match_route(
122    pattern: &str,
123    path: &str,
124) -> Option<(HashMap<String, String>, HashMap<String, String>)> {
125    let (path_only, query_opt) = split_path_query(path);
126    let pat_parts: Vec<&str> = pattern.trim_matches('/').split('/').collect();
127    let path_parts: Vec<&str> = path_only.trim_matches('/').split('/').collect();
128    if pat_parts.len() != path_parts.len() {
129        return None;
130    }
131    let mut params = HashMap::new();
132    for (pp, rp) in pat_parts.iter().zip(path_parts.iter()) {
133        if pp.starts_with(":") {
134            let name = pp.trim_start_matches(":").to_string();
135            println!(">> Param: {}\n>> Value: {}", name, rp);
136            params.insert(name, rp.to_string());
137        } else if pp != rp {
138            return None;
139        }
140    }
141
142    let query_params = parse_query(query_opt);
143
144    Some((params, query_params))
145}
